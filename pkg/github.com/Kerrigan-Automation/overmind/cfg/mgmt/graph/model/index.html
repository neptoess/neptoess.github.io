type Config ¶

A config is a structure containing parameters for something. That thing may be an application, piece of manufacturing equipment, process, etc. 

type Config struct {
    ID   string `json:"id"`
    Type string `json:"type"`
    // A human readable name for this config
    Title string `json:"title"`
    // Any additional notes related to this config
    Notes string `json:"notes"`
    // When the config was created
    CreationTime time.Time `json:"creationTime"`
    // The user that created this config
    Creator *User `json:"creator"`
    // A value indicating whether the config has been archived or not
    Archived bool   `json:"archived"`
    Contents string `json:"contents"`
}

type ConfigConsumer ¶

A config consumer is an application, machine, process, service, etc. that uses one or more configs. 

type ConfigConsumer struct {
    ID string `json:"id"`
    // Names the consumer
    Title string `json:"title"`
    // The config(s) that this consumer uses
    Configs []*ConsumedConfig `json:"configs"`
}

type ConsumedConfig ¶

A reference to a config that a consumer uses 

type ConsumedConfig struct {
    // The type of config that the consumer needs
    Type string `json:"type"`
    // A description of what the consumer uses this config for
    Description string `json:"description"`
    // The config for the consumer to use.
    // This will be null if no config is assigned.
    Config *Config `json:"config,omitempty"`
}
