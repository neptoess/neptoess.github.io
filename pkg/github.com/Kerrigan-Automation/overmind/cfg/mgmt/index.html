type Manager ¶

Manager describes an object that manages configuration data. 

type Manager interface {
    // Retrieve the config with the given ID
    Config(id string) (Config, error)

    // Retrieve all configs
    Configs() []Config

    // Create a new config
    SaveConfig(title, creator, cfgType, notes, contents string, replace bool) (Config, error)

    // Change the archive field of an existing config
    ArchiveConfig(id string, archive bool) (Config, error)

    // Retrieve all consumers
    Consumers() []Consumer

    // Create a new consumer
    NewConsumer(title string, consumed []ConsumedConfig) (Consumer, error)

    // Change the title field of an existing consumer
    UpdateConsumerTitle(id string, newTitle string) (Consumer, error)

    // Change the configs field of an existing consumer
    UpdateConsumerConfigs(id string, consumed []ConsumedConfig) (Consumer, error)

    // Delete an existing consumer.
    // Returns the deleted value.
    DeleteConsumer(id string) (Consumer, error)
}

type ManagerConfig struct {
    // What TCP port the module server should listen on
    Port uint16

    // If non-empty, use the provided directory to store config files
    ConfigDir string

    // If non-empty, sync configs with the provided git remote
    GitRemoteURL string

    // If authentication is required on the git remote, the git user name
    GitRemoteUserName string

    // If authentication is required on the git remote, the git password / personal access token
    GitRemotePassword string
}
